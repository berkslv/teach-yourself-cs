;; to10 : (listof digit) -> number
(define (to10 digits)
  (local [;; to10/acc : (listof digit) number -> number
          ;; accumulator: the base 10 representation of the digits seen so far
          (define (to10/acc digits acc)
            (cond
              [(empty? digits) acc]
              [else (to10/acc
                     (rest digits) 
                     (+ (* 10 acc) (first digits)))]))]
    (to10/acc digits 0)))

;; examples as tests
(= (to10 (list 1 0 2)) 
   102)

(= (to10 (list 2 1))
   21)

;; to10-general : number (listof digit) -> number
(define (to10-general b digits)
  (local [;; to10/acc : (listof digit) number -> number
          ;; accumulator: the base 10 representation of the digits seen so far
          (define (to10-general/acc digits acc)
            (cond
              [(empty? digits) acc]
              [else (to10-general/acc 
                     (rest digits) 
                     (+ (* b acc) (first digits)))]))]
    (to10-general/acc digits 0)))


;; examples as tests
(= (to10-general 10 (list 1 0 2)) 
   102)

(= (to10-general 8 (list 1 0 2)) 
   66)
