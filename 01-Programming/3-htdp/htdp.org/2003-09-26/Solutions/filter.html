;; below : number lon  ->  lon
;; to construct a list of those numbers
;; on alon that are below t
(define (below i l) 
  (local ((define (below-i x)
            (> i x)))
    (filter below-i l)))

;; above : lon number  ->  lon
;; to construct a list of those numbers 
;; on alon that are above t
(define (above i l) 
  (local ((define (above-i x)
            (< i x)))
    (filter above-i l)))

;; examples as tests
(equal? (below 3 empty) empty)
(equal? (below 3 (list 1 2 3 4 5))
        (list 1 2))

(equal? (above 3 empty) empty)
(equal? (above 3 (list 1 2 3 4 5))
        (list 4 5))
